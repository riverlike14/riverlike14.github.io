```cpp
#include <iostream>
using namespace std;

int root[1000001];
void _union(int a, int b);
int find(int a);

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m; cin >> n >> m;
    for (int i = 0; i <= n; i++)
        root[i] = i;
    
    int q, a, b;
    while (m--) {
        cin >> q >> a >> b;
        if (q == 0)
            _union(a, b);
        else if (find(a) == find(b))
            cout << "YES\n";
        else
            cout << "NO\n";
    }
    
    return 0;
}

void _union(int a, int b) {
    a = find(a);
    b = find(b);
    root[a] = b;
}

int find(int a) {
    if (root[root[a]] != root[a])
        root[a] = find(root[a]);
    return root[a];
}
```

```java
import java.util.StringTokenizer;
import java.io.IOException;
import java.io.BufferedReader;
import java.io.InputStreamReader;

public class Main {
    public static int[] root = new int[1000001];

    public static int find(int a) {
        if (root[root[a]] != root[a])
            root[a] = find(root[a]);
        return root[a];
    }

    public static void union(int a, int b) {
        a = find(a);
        b = find(b);
        root[a] = b;
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        StringTokenizer st = new StringTokenizer(br.readLine());
        int n = Integer.parseInt(st.nextToken());
        int m = Integer.parseInt(st.nextToken());

        for (int i = 0; i <= n; i++)
            root[i] = i;

        int q, a, b;
        StringBuilder sb = new StringBuilder();
        while (m > 0) {
            st = new StringTokenizer(br.readLine());
            q = Integer.parseInt(st.nextToken());
            a = Integer.parseInt(st.nextToken());
            b = Integer.parseInt(st.nextToken());
            
            if (q == 0)
                union(a, b);
            else if (find(a) == find(b))
                sb.append("YES\n");
            else
                sb.append("NO\n");
            m -= 1;
        }
        System.out.println(sb);
    }
}
```

```go
package main;
import (
    "fmt"
    "bufio"
    "os"
);

var root [1000001] int;

func find(a int) int {
    if root[root[a]] != root[a] {
        root[a] = find(root[a]);
    }
    return root[a];
}

func union(a, b int) {
    a, b = find(a), find(b);
    root[b] = a;
}

func main() {
    in := bufio.NewReader(os.Stdin);
    out := bufio.NewWriter(os.Stdout);
    defer out.Flush();

    var n, m int;
    fmt.Fscan(in, &n, &m);

    for i := 1; i <= n; i++ {
        root[i] = i;
    }
    
    var q, a, b int;
    for i := 1; i <= m; i++ {
        fmt.Fscan(in, &q, &a, &b);
        if q == 0 {
            union(a, b);
        } else if find(a) == find(b) {
            fmt.Fprint(out, "YES\n");
        } else {
            fmt.Fprint(out, "NO\n");
        }
    }
}
```

```rust
use std::io;
use io::{BufRead, Write};

fn find(a: usize, root: &mut Vec<usize>) -> usize {
    if root[root[a]] != root[a] {
        root[a] = find(root[a], root);
    }
    return root[a];
}

fn union(mut a: usize, mut b: usize, root: &mut Vec<usize>) -> () {
    a = find(a, root);
    b = find(b, root);
    root[a] = b;
}

fn main() {
    let stdin = io::stdin();
    let mut stdin = stdin.lock();
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    
    let mut s = String::new();
    stdin.read_line(&mut s).unwrap();

    let mut inputs = s.split_whitespace();
    let n: usize = inputs.next().unwrap().parse::<usize>().unwrap();
    let m: usize = inputs.next().unwrap().parse::<usize>().unwrap();

    let mut root: Vec<usize> = (0..=n).collect();
    
    for _ in 0..m {
        s.clear();
        stdin.read_line(&mut s).unwrap();
        let mut inputs = s.split_whitespace();
        let q: usize = inputs.next().unwrap().parse::<usize>().unwrap();
        let a: usize = inputs.next().unwrap().parse::<usize>().unwrap();
        let b: usize = inputs.next().unwrap().parse::<usize>().unwrap();

        if q == 0 {
            union(a, b, &mut root);
        } else if find(a, &mut root) == find(b, &mut root) {
            writeln!(out, "YES\n").unwrap();
        } else {
            writeln!(out, "NO\n").unwrap();
        }
    }
}
```
